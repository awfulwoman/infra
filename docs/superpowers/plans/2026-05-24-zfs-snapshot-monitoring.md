# ZFS Snapshot Monitoring Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add healthchecks.io dead-man's switches and MQTT staleness alerting to ZFS snapshot timers and belinda's pull cron so silent failures are caught within hours.

**Architecture:** Two complementary layers — healthchecks.io catches jobs that stop running entirely (via `ExecStartPost` in systemd services and a `curl` call in pull scripts); MQTT staleness alerts catch jobs that run but produce stale snapshots (via `mosquitto_pub` called from the report script and pull script). Both are opt-in per host via feature flags.

**Tech Stack:** Ansible, systemd, Python 3 (Jinja2 templates), `community.healthchecksio.checks` (already in galaxy), `mosquitto-clients` (apt), `curl` (apt, already present).

```python
def get_most_granular_interval(policy_name):
    """Return the finest interval with retention > 0 for the given policy, or None."""
    policy_def = POLICIES.get(policy_name, POLICIES['none'])
    for interval in SNAPSHOT_TYPES:
        if policy_def.get(interval, 0) > 0:
            return interval
    return None


def check_staleness(managed_datasets, all_snapshots_grouped):
    """Return (stale, healthy) lists of dataset names."""
    stale = []
    healthy = []
    now = datetime.now()

    for ds_info in managed_datasets:
        dataset = ds_info['dataset']
        policy = ds_info['policy']
        interval = get_most_granular_interval(policy)

        if interval is None:
            continue  # policy=none, not monitored

        snapshots = all_snapshots_grouped.get(dataset, [])
        interval_snaps = [s for s in snapshots if parse_snapshot(s) and parse_snapshot(s)[1] == interval]

        if not interval_snaps:
            stale.append(dataset)
            continue

        newest_ts_str, _ = parse_snapshot(interval_snaps[-1])
        newest_dt = datetime.strptime(newest_ts_str, "%Y-%m-%d_%H:%M:%S")

        if (now - newest_dt) > STALE_THRESHOLDS[interval]:
            stale.append(dataset)
        else:
            healthy.append(dataset)

    return stale, healthy


def publish_mqtt(host_name, stale, healthy, mqtt_host, mqtt_topic):
    """Publish snapshot status to MQTT via mosquitto_pub."""
    payload = json.dumps({
        "ok": len(stale) == 0,
        "host": host_name,
        "checked_at": datetime.now().strftime("%Y-%m-%dT%H:%M:%S"),
        "stale": stale,
        "healthy": healthy,
    })
    cmd = ["mosquitto_pub", "-h", mqtt_host, "-t", mqtt_topic, "-m", payload]
    try:
        result = subprocess.run(cmd, capture_output=True, timeout=10, check=False)
        if result.returncode != 0:
            error(f"mosquitto_pub failed: {result.stderr.decode().strip()}")
    except Exception as e:
        error(f"Failed to publish MQTT: {e}")
```

```python
def get_newest_autosnap_time(dataset):
    """Return the datetime of the newest autosnap snapshot for a local dataset, or None."""
    snapshots = get_local_snapshots(dataset)
    autosnap_snaps = [s for s in snapshots if s.startswith('autosnap_')]
    if not autosnap_snaps:
        return None
    newest = autosnap_snaps[-1]
    match = re.match(r'^autosnap_(\d{4}-\d{2}-\d{2}_\d{2}:\d{2}:\d{2})_', newest)
    if match:
        return datetime.strptime(match.group(1), "%Y-%m-%d_%H:%M:%S")
    return None


def publish_mqtt_status(name, datasets, destination, mqtt_host, mqtt_topic_prefix, stale_multiplier=2):
    """Publish pull status to MQTT after a successful pull."""
    now = datetime.now()
    stale_threshold = timedelta(hours=stale_multiplier * 2)

    stale_datasets = []
    healthy_datasets = []
    for dataset in datasets:
        local_ds = f"{destination}/{name}/{dataset}"
        newest_time = get_newest_autosnap_time(local_ds)
        if newest_time is None or (now - newest_time) > stale_threshold:
            stale_datasets.append(dataset)
        else:
            healthy_datasets.append(dataset)

    topic = f"{mqtt_topic_prefix}/{name}/backups"
    payload = json.dumps({
        "ok": len(stale_datasets) == 0,
        "host": name,
        "pulled_at": now.strftime("%Y-%m-%dT%H:%M:%S"),
        "stale_datasets": stale_datasets,
        "healthy_datasets": healthy_datasets,
    })

    cmd = ["mosquitto_pub", "-h", mqtt_host, "-t", topic, "-m", payload]
    try:
        result = subprocess.run(cmd, capture_output=True, timeout=10, check=False)
        if result.returncode != 0:
            error(f"mosquitto_pub failed: {result.stderr.decode().strip()}")
        else:
            debug(f"Published MQTT status to {topic}")
    except Exception as e:
        error(f"Failed to publish MQTT status: {e}")
```
